    <script>
        let currentQR = null;
        const generateBtn = document.getElementById('generateBtn');
        const generateText = document.getElementById('generateText');
        
        function generateQR() {
            const url = document.getElementById('urlInput').value.trim();
            
            if (!url) {
                showError('Por favor ingresa una URL válida');
                return;
            }
            
            // Validar formato de URL
            try {
                new URL(url);
            } catch (e) {
                showError('La URL no es válida. Asegúrate de incluir http:// o https://');
                return;
            }
            
            // Mostrar estado de carga
            generateBtn.disabled = true;
            generateText.innerHTML = '<div class="spinner"></div> Generando...';
            
            // Limpiar QR anterior después de un breve retraso para la animación
            setTimeout(() => {
                document.getElementById('qrContainer').innerHTML = '';
                
                // Crear contenedor para la animación de entrada
                const qrDiv = document.createElement('div');
                qrDiv.id = 'qrcode';
                qrDiv.className = 'fade-in';
                document.getElementById('qrContainer').appendChild(qrDiv);
                
                // Generar nuevo QR
                currentQR = new QRCode(qrDiv, {
                    text: url,
                    width: 256,
                    height: 256,
                    colorDark: "#4361ee",  // Color primario
                    colorLight: "#ffffff",
                    correctLevel: QRCode.CorrectLevel.H
                });
                
                // Habilitar botón de descarga
                document.getElementById('downloadBtn').disabled = false;
                
                // Restaurar botón de generación
                generateBtn.disabled = false;
                generateText.textContent = 'Generar Código QR';
            }, 500);
        }
        
        function downloadQR() {
            if (!currentQR) {
                showError('Primero genera un código QR');
                return;
            }
            
            const canvas = document.querySelector('#qrcode canvas');
            if (!canvas) {
                showError('No se pudo generar la imagen para descargar');
                return;
            }
            
            // Crear enlace de descarga
            const link = document.createElement('a');
            link.download = 'codigo-qr-' + new Date().toISOString().slice(0,10) + '.png';
            link.href = canvas.toDataURL('image/png');
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
        }
        
        function showError(message) {
            const errorDiv = document.createElement('div');
            errorDiv.style.color = 'var(--error-color)';
            errorDiv.style.margin = '10px 0';
            errorDiv.style.fontWeight = '500';
            errorDiv.textContent = message;
            
            const inputGroup = document.querySelector('.input-group');
            const existingError = inputGroup.querySelector('.error-message');
            if (existingError) {
                inputGroup.removeChild(existingError);
            }
            
            errorDiv.className = 'error-message fade-in';
            inputGroup.appendChild(errorDiv);
            
            // Restaurar botón si había error
            generateBtn.disabled = false;
            generateText.textContent = 'Generar Código QR';
        }
        
        // Generar QR al presionar Enter
        document.getElementById('urlInput').addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                generateQR();
            }
        });
    </script>
